# Required packages when installing Qt5 from source on Ubuntu:

```
# for xcb platform
libx11-dev
libx11-xcb-dev
libxext-dev
libxfixes-dev
libxi-dev
libxrender-dev
libxcb1-dev
libxcb-glx0-dev
libxcb-keysyms1-dev
libxcb-image0-dev
libxcb-shm0-dev
libxcb-icccm4-dev
libxcb-sync0-dev
libxcb-xfixes0-dev
libxcb-shape0-dev
libxcb-randr0-dev
libxcb-render-util0-dev
libxcd-xinerama-dev (should be libxcb-xinerama0-dev on ubuntu 18.04)
libxkbcommon-dev
libxkbcommon-x11-dev

# optional for simple build, required for AppImage (needed for audio feedback)
libpulse-dev
libalsaplayer-dev
```
